account::create_account()

void create_account();

Gets data from the user for the user account.

Parameters

none

Return value

none

Example

#include < iostream >
#include < fstream >
#include < cctype >
using namespace std;

void write_account() {
   account ac;
   ac.create_account();
   ofstream outFile;
   outFile.open("account.dat",ios::binary|ios::app);
   outFile.write(reinterpret_cast (&ac), sizeof(account));
   outFile.close();
   }

void write_account();

int main () {

   char choice;
   cout << "Press "1" to create a new bank account" << endl;

   if (choice == 1){
      write_account();
   }
   else {
      return 0;
   }
}
